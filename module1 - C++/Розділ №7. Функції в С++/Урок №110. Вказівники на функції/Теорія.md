# Урок №110. Вказівники на функції

Вказівник на функцію зберігає адресу функції, дозволяючи викликати її через вказівник.

## Синтаксис

```c++
тип_повернення (*ім'я_вказівника)(параметри);
```

## Приклад

```c++
#include <iostream>

int add(int a, int b) { return a + b; }
int subtract(int a, int b) { return a - b; }

int main()
{
    int (*operation)(int, int);  // вказівник на функцію
    
    operation = add;             // вказує на add
    std::cout << operation(10, 3) << "\n";  // 13
    
    operation = subtract;        // вказує на subtract
    std::cout << operation(10, 3) << "\n";  // 7
    
    return 0;
}
```

## Тип через using

```c++
using Operation = int(*)(int, int);

int add(int a, int b) { return a + b; }

void execute(Operation op, int a, int b)
{
    std::cout << op(a, b) << "\n";
}

int main()
{
    execute(add, 5, 3);  // 8
    return 0;
}
```

## Масив вказівників на функції

```c++
int add(int a, int b) { return a + b; }
int sub(int a, int b) { return a - b; }
int mul(int a, int b) { return a * b; }

int main()
{
    using Op = int(*)(int, int);
    Op operations[] = {add, sub, mul};
    
    for (auto op : operations)
        std::cout << op(10, 5) << " ";  // 15 5 50
    
    return 0;
}
```

## Сучасна альтернатива: std::function

```c++
#include <functional>

int add(int a, int b) { return a + b; }

int main()
{
    std::function<int(int, int)> func = add;
    std::cout << func(5, 3);  // 8
    
    // Також лямбди:
    func = [](int a, int b) { return a * b; };
    std::cout << func(5, 3);  // 15
    
    return 0;
}
```

## Правило

Вказівники на функції корисні для callback-ів. У сучасному C++ віддавайте перевагу `std::function` та лямбдам.
