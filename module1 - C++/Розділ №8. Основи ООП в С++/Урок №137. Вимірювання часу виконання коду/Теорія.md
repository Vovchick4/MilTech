# Урок №137. Вимірювання часу виконання (таймінг) коду

Вимірювання часу допомагає знаходити "вузькі місця" в програмі.

## Базовий спосіб: <chrono> (C++11)

```c++
#include <iostream>
#include <chrono>

int main()
{
    // Початок
    auto start = std::chrono::high_resolution_clock::now();
    
    // Код для вимірювання
    int sum = 0;
    for (int i = 0; i < 1000000; ++i)
        sum += i;
    
    // Кінець
    auto end = std::chrono::high_resolution_clock::now();
    
    // Різниця
    auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
    
    std::cout << "Час: " << duration.count() << " мс\n";
    std::cout << "Сума: " << sum << "\n";
    
    return 0;
}
```

## Клас для таймінгу

```c++
#include <iostream>
#include <chrono>

class Timer
{
    using Clock = std::chrono::high_resolution_clock;
    using Ms = std::chrono::milliseconds;
    
    std::chrono::time_point<Clock> start;
    std::string name;
    
public:
    Timer(const std::string& n) : name{n}
    {
        start = Clock::now();
    }
    
    ~Timer()
    {
        auto end = Clock::now();
        auto ms = std::chrono::duration_cast<Ms>(end - start).count();
        std::cout << name << ": " << ms << " мс\n";
    }
};

// Використання:
void someFunction()
{
    Timer t("someFunction");
    // ... код ...
}  // деструктор виведе час
```

## Порівняння векторів

```c++
#include <iostream>
#include <vector>
#include <chrono>

int main()
{
    const int N = 1000000;
    
    {
        Timer t("Без reserve");
        std::vector<int> v;
        for (int i = 0; i < N; ++i)
            v.push_back(i);
    }
    
    {
        Timer t("З reserve");
        std::vector<int> v;
        v.reserve(N);
        for (int i = 0; i < N; ++i)
            v.push_back(i);
    }
    
    return 0;
}
```

## Використання: обчислення π

```c++
#include <iostream>
#include <chrono>
#include <cmath>

int main()
{
    auto start = std::chrono::steady_clock::now();
    
    const int N = 100000000;
    double pi = 0.0;
    for (int i = 0; i < N; ++i)
        pi += (i % 2 == 0 ? 1.0 : -1.0) / (2 * i + 1);
    pi *= 4;
    
    auto end = std::chrono::steady_clock::now();
    auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(end - start).count();
    
    std::cout << "π ≈ " << pi << "\n";
    std::cout << "Час: " << ms << " мс\n";
    
    return 0;
}
```

## Нанoseconds

```c++
#include <chrono>

auto start = std::chrono::high_resolution_clock::now();
// швидкий код
auto end = std::chrono::high_resolution_clock::now();

auto ns = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start).count();
std::cout << ns << " нс\n";
```
