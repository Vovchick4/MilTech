# Урок №148. Перевантаження операторів введення/виведення

Оператори `<<` (виведення) і `>>` (введення) для роботи з потоками.

## Виведення <<

Дружня функція (не метод!):

```c++
class Point
{
    int x, y;
    
public:
    Point(int x, int y) : x{x}, y{y} {}
    
    friend std::ostream& operator<<(std::ostream& os, const Point& p)
    {
        os << "(" << p.x << ", " << p.y << ")";
        return os;  // обов'язково повернути потік
    }
};

int main()
{
    Point p(3, 4);
    std::cout << p;           // (3, 4)
    std::cout << p << "\n";   // ланцюжок: (3, 4)
    return 0;
}
```

## Введення >>

```c++
class Point
{
    int x, y;
    
public:
    Point() = default;
    
    friend std::istream& operator>>(std::istream& is, Point& p)
    {
        is >> p.x >> p.y;
        return is;
    }
    
    friend std::ostream& operator<<(std::ostream& os, const Point& p)
    {
        os << "(" << p.x << ", " << p.y << ")";
        return os;
    }
};

int main()
{
    Point p;
    std::cout << "Введіть x y: ";
    std::cin >> p;        // введення
    std::cout << p;       // виведення
    return 0;
}
```

## Форматування

```c++
class Time
{
    int hours, minutes;
    
public:
    Time(int h, int m) : hours{h}, minutes{m} {}
    
    friend std::ostream& operator<<(std::ostream& os, const Time& t)
    {
        os << t.hours << ":";
        if (t.minutes < 10) os << "0";
        os << t.minutes;
        return os;
    }
};

int main()
{
    Time t(9, 5);
    std::cout << t;  // 9:05
    return 0;
}
```

## Приклад: комплексне число

```c++
class Complex
{
    double real, imag;
    
public:
    Complex(double r = 0, double i = 0) : real{r}, imag{i} {}
    
    friend std::ostream& operator<<(std::ostream& os, const Complex& c)
    {
        os << c.real;
        if (c.imag >= 0)
            os << "+" << c.imag << "i";
        else
            os << "-" << -c.imag << "i";
        return os;
    }
    
    friend std::istream& operator>>(std::istream& is, Complex& c)
    {
        // формат: real imag
        is >> c.real >> c.imag;
        return is;
    }
};
```

## Через методи (для << у підкласах)

```c++
class Printable
{
public:
    virtual void print(std::ostream& os) const = 0;
    
    friend std::ostream& operator<<(std::ostream& os, const Printable& p)
    {
        p.print(os);
        return os;
    }
};
```

## Правила

- Оператор `<<` — дружня функція, не метод
- Завжди повертайте `os` для ланцюжків
- Для `>>` передавайте не-const посилання
