# Урок №140. Перевантаження операторів порівняння

Оператори ==, !=, <, >, <=, >= для власних типів.

## Приклад: клас Point

```c++
class Point
{
    int x, y;
    
public:
    Point(int x, int y) : x{x}, y{y} {}
    
    bool operator==(const Point& other) const
    {
        return x == other.x && y == other.y;
    }
    
    bool operator!=(const Point& other) const
    {
        return !(*this == other);  // через ==
    }
    
    bool operator<(const Point& other) const
    {
        if (x != other.x) return x < other.x;
        return y < other.y;
    }
    
    bool operator>(const Point& other) const
    {
        return other < *this;  // через <
    }
    
    bool operator<=(const Point& other) const
    {
        return !(other < *this);  // через <
    }
    
    bool operator>=(const Point& other) const
    {
        return !(*this < other);  // через <
    }
};
```

## Через дружні функції

```c++
class Point
{
    int x, y;
    
public:
    Point(int x, int y) : x{x}, y{y} {}
    
    friend bool operator==(const Point& a, const Point& b)
    {
        return a.x == b.x && a.y == b.y;
    }
    
    friend bool operator<(const Point& a, const Point& b)
    {
        if (a.x != b.x) return a.x < b.x;
        return a.y < b.y;
    }
};
```

## Приклад: клас Person

```c++
class Person
{
    std::string name;
    int age;
    
public:
    Person(const std::string& name, int age)
        : name{name}, age{age} {}
    
    bool operator==(const Person& other) const
    {
        return name == other.name && age == other.age;
    }
    
    bool operator<(const Person& other) const
    {
        if (name != other.name) return name < other.name;
        return age < other.age;
    }
};
```

## spaceship оператор <=> (C++20)

```c++
#include <compare>

class Point
{
    int x, y;
    
public:
    auto operator<=>(const Point& other) const = default;
    // Автоматично генерує ==, !=, <, >, <=, >=
};

// Або власна реалізація:
class Person
{
    std::string name;
    int age;
    
public:
    auto operator<=>(const Person& other) const
    {
        if (auto cmp = name <=> other.name; cmp != 0)
            return cmp;
        return age <=> other.age;
    }
};
```

## Використання в контейнерах

```c++
#include <set>
#include <map>

int main()
{
    std::set<Person> people;  // потребує operator<
    std::map<Person, int> scores;  // потребує operator<
    
    std::vector<Person> vec;
    std::sort(vec.begin(), vec.end());  // потребує operator<
    return 0;
}
```

## Правило

Мінімально реалізуйте == і <, інші виводьте через них. У C++20 використовуйте <=>.
