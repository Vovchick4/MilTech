# Урок №149. Перевантаження операторів перетворення типів

Дозволяє визначити, як ваш тип перетворюється на інші типи.

## operator type()

```c++
class Rational
{
    int num, den;
    
public:
    Rational(int n, int d) : num{n}, den{d} {}
    
    operator double() const  // Rational → double
    {
        return static_cast<double>(num) / den;
    }
    
    operator int() const  // Rational → int (відкидає дробову частину)
    {
        return num / den;
    }
};

int main()
{
    Rational r(3, 4);
    
    double d = r;  // 0.75
    int i = r;     // 0 (3/4 = 0)
    
    std::cout << d << " " << i;  // 0.75 0
    
    return 0;
}
```

## explicit (C++11)

Запобігає неявному перетворенню:

```c++
class SafeBool
{
    bool value;
    
public:
    explicit SafeBool(bool v) : value{v} {}
    
    explicit operator bool() const  // explicit — тільки явне перетворення
    {
        return value;
    }
};

int main()
{
    SafeBool b(true);
    
    // if (b)        // OK: контекст bool
    // bool x = b;   // ПОМИЛКА! explicit
    bool x = static_cast<bool>(b);  // OK
    
    return 0;
}
```

## Приклад: String

```c++
class MyString
{
    char* str;
    
public:
    MyString(const char* s = "")
    {
        str = new char[std::strlen(s) + 1];
        std::strcpy(str, s);
    }
    
    ~MyString() { delete[] str; }
    
    operator const char*() const  // MyString → const char*
    {
        return str;
    }
};

void print(const char* s) { std::cout << s; }

int main()
{
    MyString s("Hello");
    print(s);  // неявне перетворення MyString → const char*
    return 0;
}
```

## Користувацькі літерали (C++11)

```c++
class Distance
{
    double meters;
    
public:
    explicit Distance(double m) : meters{m} {}
    
    double getMeters() const { return meters; }
    double getKm() const { return meters / 1000.0; }
};

Distance operator"" _km(long double km)
{
    return Distance(km * 1000.0);
}

Distance operator"" _m(long double m)
{
    return Distance(m);
}

int main()
{
    Distance d1 = 5.0_km;  // 5000 м
    Distance d2 = 300.0_m;  // 300 м
    
    std::cout << d1.getMeters();  // 5000
    return 0;
}
```

## Використання в switch

```c++
class Status
{
    int code;
    
public:
    Status(int c) : code{c} {}
    
    explicit operator int() const { return code; }
};

int main()
{
    Status s(2);
    
    switch (static_cast<int>(s))
    {
        case 1: break;
        case 2: break;
    }
    
    return 0;
}
```

## Правило

Робіть оператори перетворення `explicit`, якщо невпевнені. Це запобігає несподіваним перетворенням.
