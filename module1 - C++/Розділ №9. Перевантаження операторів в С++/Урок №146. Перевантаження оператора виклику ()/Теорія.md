# Урок №146. Перевантаження оператора виклику ()

Оператор `()` дозволяє використовувати об'єкт як функцію (функтор).

## Синтаксис

```c++
class Multiplier
{
    int factor;
    
public:
    Multiplier(int f) : factor{f} {}
    
    int operator()(int x) const
    {
        return x * factor;
    }
};

int main()
{
    Multiplier doubler(2);
    
    std::cout << doubler(5);  // 10 — об'єкт як функція!
    std::cout << doubler(10); // 20
    
    return 0;
}
```

## Різні сигнатури

```c++
class Operation
{
public:
    // Різні варіанти виклику
    int operator()(int a, int b) const { return a + b; }
    double operator()(double a, double b) const { return a + b; }
    int operator()(int a, int b, int c) const { return a + b + c; }
};
```

## Приклад: калькулятор

```c++
class Calculator
{
public:
    int operator()(int a, int b, char op) const
    {
        switch (op)
        {
            case '+': return a + b;
            case '-': return a - b;
            case '*': return a * b;
            case '/': return b != 0 ? a / b : 0;
            default:  return 0;
        }
    }
};

int main()
{
    Calculator calc;
    std::cout << calc(10, 5, '+');  // 15
    std::cout << calc(10, 5, '-');  // 5
    std::cout << calc(10, 5, '*');  // 50
    std::cout << calc(10, 5, '/');  // 2
    return 0;
}
```

## Функтори в алгоритмах

```c++
#include <algorithm>
#include <vector>

class GreaterThan
{
    int threshold;
    
public:
    GreaterThan(int t) : threshold{t} {}
    
    bool operator()(int x) const
    {
        return x > threshold;
    }
};

int main()
{
    std::vector<int> v = {1, 5, 8, 12, 3, 9};
    
    auto it = std::find_if(v.begin(), v.end(), GreaterThan(7));
    // *it = 8
    
    // Через лямбду:
    // auto it = std::find_if(v.begin(), v.end(), [](int x){ return x > 7; });
    
    return 0;
}
```

## Приклад: сортування

```c++
struct Person
{
    std::string name;
    int age;
};

class CompareByAge
{
public:
    bool operator()(const Person& a, const Person& b) const
    {
        return a.age < b.age;
    }
};

int main()
{
    std::vector<Person> people = {{"Анна", 25}, {"Iван", 20}, {"Марiя", 30}};
    
    std::sort(people.begin(), people.end(), CompareByAge());
    // Iван, Анна, Марiя
    
    return 0;
}
```

## Зберігання стану

```c++
class Counter
{
    int count = 0;
    
public:
    int operator()()
    {
        return ++count;
    }
    
    void reset() { count = 0; }
};

int main()
{
    Counter counter;
    std::cout << counter();  // 1
    std::cout << counter();  // 2
    std::cout << counter();  // 3
    counter.reset();
    std::cout << counter();  // 1
    return 0;
}
```

## Правило

Використовуйте `operator()` для створення функторів — об'єктів, що поводяться як функції, але можуть зберігати стан.
