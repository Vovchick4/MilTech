# Урок №152. Метод проти дружньої функції

Деякі оператори краще реалізувати як методи, інші — як дружні функції.

## Порівняння

| Аспект | Метод класу | Дружня функція |
|--------|-------------|----------------|
| Лівий операнд | обов'язково *this | будь-який тип |
| Неявне перетворення | тільки для правого | для обох |
| Доступ до private | так | так (через friend) |
| Синтаксис | всередині класу | зовні |

## Оператори, які мають бути методами

```c++
class MyClass
{
public:
    // Змінюють об'єкт — однозначно методи
    MyClass& operator=(const MyClass&);
    MyClass& operator+=(const MyClass&);
    MyClass& operator-=(const MyClass&);
    
    // Доступ до об'єкта
    T& operator[](int);
    const T& operator[](int) const;
    type operator()(...);
    T* operator->();
    
    // Перетворення
    operator type() const;
    
    // Розіменування
    T& operator*();
};
```

## Оператори як методи або функції

```c++
class Rational
{
    int num, den;
    
public:
    // Як метод:
    Rational operator+(const Rational& other) const
    {
        return {num * other.den + other.num * den, den * other.den};
    }
    
    // Проблема: int + Rational не працює
    // Rational r = 5 + Rational(1, 2);  // ПОМИЛКА!
};
```

## Як дружня функція (краще)

```c++
class Rational
{
    int num, den;
    
public:
    Rational(int n, int d = 1) : num{n}, den{d} {}
    
    friend Rational operator+(const Rational& a, const Rational& b)
    {
        return {a.num * b.den + b.num * a.den, a.den * b.den};
    }
};

int main()
{
    Rational a(1, 2);
    
    Rational r1 = a + Rational(1, 3);  // OK
    Rational r2 = Rational(1, 3) + a;  // OK
    Rational r3 = a + 5;               // OK (5 → Rational(5))
    Rational r4 = 5 + a;               // OK (5 → Rational(5))
    
    return 0;
}
```

## Схема вибору

```c++
class Number
{
    int value;
    
public:
    // Унарні — методи
    Number operator-() const;
    Number& operator++();
    
    // += -= — методи
    Number& operator+=(const Number&);
    
    // Бінарні — дружні функції (для симетрії)
    friend Number operator+(const Number& a, const Number& b);
    friend bool operator==(const Number& a, const Number& b);
    
    // << >> — обов'язково дружні функції
    friend std::ostream& operator<<(std::ostream&, const Number&);
};
```

## Приклад: компроміс

```c++
class Vector2D
{
    double x, y;
    
public:
    // += як метод (змінює об'єкт)
    Vector2D& operator+=(const Vector2D& other)
    {
        x += other.x; y += other.y; return *this;
    }
    
    // + як дружня функція (симетрично)
    friend Vector2D operator+(Vector2D a, const Vector2D& b)
    {
        a += b;  // використовує +=
        return a;
    }
};
```

## Правило вибору

- **Метод**: =, +=, -=, *=, /=, [], (), ->, *, перетворення типів
- **Дружня функція**: <<, >>, бінарні арифметичні, порівняння
