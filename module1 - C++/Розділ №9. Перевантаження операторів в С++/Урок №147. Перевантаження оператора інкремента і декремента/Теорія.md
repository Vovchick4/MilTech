# Урок №147. Перевантаження інкремента і декремента

++ і -- мають префіксну (++x) і постфіксну (x++) форми.

## Синтаксис

```c++
class Counter
{
    int value;
    
public:
    // Префікс: ++c
    Counter& operator++()
    {
        ++value;
        return *this;
    }
    
    // Постфікс: c++ (фіктивний параметр int)
    Counter operator++(int)
    {
        Counter temp = *this;
        ++value;
        return temp;  // повертає старе значення
    }
    
    // Декремент
    Counter& operator--()
    {
        --value;
        return *this;
    }
    
    Counter operator--(int)
    {
        Counter temp = *this;
        --value;
        return temp;
    }
    
    int getValue() const { return value; }
};
```

## Використання

```c++
int main()
{
    Counter c;
    
    ++c;  // префікс: збільшує і повертає посилання
    c++;  // постфікс: повертає копію до збільшення
    
    Counter d = ++c;  // d = нове значення
    Counter e = c++;  // e = старе значення
    
    return 0;
}
```

## Ітератор

```c++
class Iterator
{
    int* ptr;
    
public:
    Iterator(int* p) : ptr{p} {}
    
    Iterator& operator++()  // префікс
    {
        ++ptr;
        return *this;
    }
    
    Iterator operator++(int)  // постфікс
    {
        Iterator old = *this;
        ++ptr;
        return old;
    }
    
    int& operator*() { return *ptr; }
    bool operator!=(const Iterator& other) { return ptr != other.ptr; }
};
```

## Приклад: крок

```c++
class StepCounter
{
    int value;
    int step;
    
public:
    StepCounter(int v, int s) : value{v}, step{s} {}
    
    StepCounter& operator++()
    {
        value += step;
        return *this;
    }
    
    StepCounter operator++(int)
    {
        StepCounter old = *this;
        value += step;
        return old;
    }
    
    int getValue() const { return value; }
};

int main()
{
    StepCounter sc(0, 5);
    ++sc;  // 5
    ++sc;  // 10
    sc++;  // повертає 10, стає 15
    return 0;
}
```

## Продуктивність

Префікс ++ зазвичай швидший (не створює копію):

```c++
// Префікс (рекомендується):
for (auto it = vec.begin(); it != vec.end(); ++it)

// Постфікс (копіює):
for (auto it = vec.begin(); it != vec.end(); it++)
```

## Правило

Префікс повертає `*this` (посилання), постфікс — копію до зміни. Для вбудованих типів різниці немає.
