# Урок №145. Перевантаження оператора індексування []

Оператор `[]` дозволяє доступатися до елементів власного контейнера за індексом.

## Синтаксис

```c++
class Array
{
    int data[100];
    
public:
    int& operator[](int index)
    {
        return data[index];
    }
    
    const int& operator[](int index) const  // для const об'єктів
    {
        return data[index];
    }
};
```

## Приклад: простий вектор

```c++
class IntVector
{
    int* data;
    int size;
    
public:
    IntVector(int s) : size{s}
    {
        data = new int[size]{};
    }
    
    ~IntVector() { delete[] data; }
    
    int& operator[](int index)
    {
        if (index < 0 || index >= size)
            throw std::out_of_range("Index out of range");
        return data[index];
    }
    
    const int& operator[](int index) const
    {
        if (index < 0 || index >= size)
            throw std::out_of_range("Index out of range");
        return data[index];
    }
    
    int getSize() const { return size; }
};

int main()
{
    IntVector v(10);
    v[0] = 42;
    v[5] = 100;
    
    for (int i = 0; i < v.getSize(); ++i)
        std::cout << v[i] << " ";
    
    const IntVector& cv = v;
    // cv[0] = 5;  // ПОМИЛКА! const версія повертає const&
    std::cout << cv[0];  // OK: читання
    
    return 0;
}
```

## Приклад: матриця 2D

```c++
class Matrix
{
    int** data;
    int rows, cols;
    
public:
    Matrix(int r, int c) : rows{r}, cols{c}
    {
        data = new int*[rows];
        for (int i = 0; i < rows; ++i)
            data[i] = new int[cols]{};
    }
    
    ~Matrix()
    {
        for (int i = 0; i < rows; ++i)
            delete[] data[i];
        delete[] data;
    }
    
    int* operator[](int row)
    {
        return data[row];  // повертає рядок
    }
    
    const int* operator[](int row) const
    {
        return data[row];
    }
};

int main()
{
    Matrix m(3, 4);
    m[1][2] = 42;  // m.operator[](1)[2]
    std::cout << m[1][2];  // 42
    return 0;
}
```

## Приклад: словник

```c++
class Dictionary
{
    struct Entry { std::string key; int value; };
    std::vector<Entry> entries;
    
public:
    int& operator[](const std::string& key)
    {
        for (auto& e : entries)
            if (e.key == key) return e.value;
        
        entries.push_back({key, 0});
        return entries.back().value;
    }
};
```

## Діапазонний for

```c++
class Range
{
    int start, end;
    
public:
    Range(int s, int e) : start{s}, end{e} {}
    
    int operator[](int index) const
    {
        return start + index;
    }
    
    int size() const { return end - start; }
};
```

## Правило

Оператор `[]` має повертати посилання (для читання і запису). Для безпеки додавайте перевірку меж.
