# Урок №141. Перевантаження оператора присвоювання

`operator=` копіює дані з одного об'єкта в інший. Компілятор генерує його автоматично, але для динамічних ресурсів потрібен власний.

## Синтаксис

```c++
class MyClass
{
    Type* data;
    
public:
    MyClass& operator=(const MyClass& other)
    {
        // 1. Захист від самоприсвоєння
        if (this == &other)
            return *this;
        
        // 2. Звільнити старі ресурси
        delete[] data;
        
        // 3. Скопіювати нові дані
        data = new Type[*other.data];
        // ... копіювання ...
        
        // 4. Повернути *this
        return *this;
    }
};
```

## Повний приклад

```c++
class MyString
{
    char* str;
    int length;
    
public:
    MyString(const char* s = "")
    {
        length = std::strlen(s);
        str = new char[length + 1];
        std::strcpy(str, s);
    }
    
    ~MyString()
    {
        delete[] str;
    }
    
    // Копіювальне присвоювання
    MyString& operator=(const MyString& other)
    {
        if (this == &other)
            return *this;
        
        delete[] str;
        
        length = other.length;
        str = new char[length + 1];
        std::strcpy(str, other.str);
        
        return *this;
    }
    
    void print() const { std::cout << str; }
};
```

## Використання

```c++
int main()
{
    MyString a("Hello");
    MyString b("World");
    
    b = a;  // оператор присвоювання
    
    MyString c = a;  // це конструктор копіювання, не присвоювання!
    
    return 0;
}
```

## Copy-and-swap ідіома

Кращий спосіб реалізувати operator=:

```c++
class MyString
{
    char* str;
    int length;
    
    void swap(MyString& other) noexcept
    {
        std::swap(str, other.str);
        std::swap(length, other.length);
    }
    
public:
    MyString(const char* s = "") : str{nullptr}, length{0}
    {
        length = std::strlen(s);
        str = new char[length + 1];
        std::strcpy(str, s);
    }
    
    MyString(const MyString& other) : MyString(other.str) {}
    
    MyString& operator=(MyString other)  // копія через передачу за значенням
    {
        swap(other);  // обмін
        return *this;
    }
    
    ~MyString() { delete[] str; }
};
```

## Присвоєння з переміщенням (C++11)

```c++
class MyString
{
public:
    MyString& operator=(MyString&& other) noexcept
    {
        if (this != &other)
        {
            delete[] str;
            str = other.str;
            length = other.length;
            other.str = nullptr;
            other.length = 0;
        }
        return *this;
    }
};
```

## Правило

Operator= має: 1) перевіряти самоприсвоєння, 2) звільняти старі ресурси, 3) копіювати нові, 4) повертати *this.
