# Урок №151. Перевантаження оператора стрілка ->

Оператор `->` дозволяє класу поводитися як розумний вказівник.

## Синтаксис

```c++
class Ptr
{
    Object* obj;
    
public:
    Ptr(Object* p) : obj{p} {}
    
    Object* operator->()  // повертає вказівник
    {
        return obj;
    }
    
    Object& operator*()   // розіменування
    {
        return *obj;
    }
};
```

## Приклад: розумний вказівник (спрощений)

```c++
template<typename T>
class SmartPtr
{
    T* ptr;
    
public:
    explicit SmartPtr(T* p = nullptr) : ptr{p} {}
    
    ~SmartPtr() { delete ptr; }
    
    T* operator->()
    {
        return ptr;
    }
    
    const T* operator->() const
    {
        return ptr;
    }
    
    T& operator*()
    {
        return *ptr;
    }
    
    const T& operator*() const
    {
        return *ptr;
    }
    
    // Заборона копіювання
    SmartPtr(const SmartPtr&) = delete;
    SmartPtr& operator=(const SmartPtr&) = delete;
    
    // Переміщення
    SmartPtr(SmartPtr&& other) noexcept : ptr{other.ptr}
    {
        other.ptr = nullptr;
    }
};

struct Point
{
    int x, y;
    void print() const { std::cout << x << "," << y; }
};

int main()
{
    SmartPtr<Point> sp(new Point{3, 4});
    
    sp->x = 5;           // operator->
    sp->print();         // operator->
    (*sp).y = 10;        // operator*
    
    return 0;
}
```

## Ланцюжковий виклик

Оператор `->` викликається рекурсивно:

```c++
class A { public: void func() {} };
class B
{
    A* a;
public:
    B(A* p) : a{p} {}
    A* operator->() { return a; }
};
class C
{
    B b;
public:
    C(A* p) : b{p} {}
    B& operator->() { return b; }
};

int main()
{
    A a;
    C c(&a);
    c->func();  // C::operator-> → B::operator-> → A::func
    return 0;
}
```

## Приклад: ітератор

```c++
template<typename T>
class Vector
{
    T* data;
    int size;
    
public:
    class Iterator
    {
        T* ptr;
    public:
        Iterator(T* p) : ptr{p} {}
        
        T& operator*() { return *ptr; }
        T* operator->() { return ptr; }
        
        Iterator& operator++() { ++ptr; return *this; }
        bool operator!=(const Iterator& other) { return ptr != other.ptr; }
    };
    
    Iterator begin() { return Iterator(data); }
    Iterator end()   { return Iterator(data + size); }
};

struct Person { std::string name; int age; };

int main()
{
    Vector<Person> people;
    
    for (auto it = people.begin(); it != people.end(); ++it)
    {
        std::cout << it->name;  // operator->
        std::cout << (*it).age; // operator*
    }
    
    return 0;
}
```

## Обмеження

- `->` має повертати вказівник або інший об'єкт з перевантаженим `->`
- `->` — унарний оператор (не приймає параметрів)

## Правило

Перевантажуйте `->` для класів, які мають поводитися як вказівники: розумні вказівники, ітератори, обгортки.
